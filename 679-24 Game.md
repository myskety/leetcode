## 问题描述


You have 4 cards each containing a number fron 1 to 9. You need to judge whether they could operated through *, /, +, -, (, ) to get the value of 24.</br>
<br>

#### Example 1
<pre>
Input: [4, 1, 8, 7] </br>
Output: true</br>
Explanation: (8-1) * (7-1) = 24
</pre>
#### Example 2
<pre>
Input: [1, 2, 1, 2] </br>
Output: false</br>
</pre>
Explanation: </br>

24点游戏，判断是否有解。


#### Recursive Solution

* 递归遍历列表里的任意两个数字(p, q)，其运算结果为<code>p+q, p-q, p*q, p/q(q != 0), q/p(p != 0)</code>。

* 将上面的结果依次加入到列表中，这样列表中元素个数减一，即可对新列表进行递归。

* 当列表中元素个数为一时，判断该元素与24是否相等，返回结果。

#### AC code

```

class Solution {
public:
    bool judgePoint24(vector<int>& nums) {
        bool ans = false;
        double eps = 0.0001; // double数据的判等条件
        vector<double> arr(nums.begin(), nums.end());
    }

    void helper(vector<double>& nums, double eps, bool& ans){
        if(ans)
            return;
        if(nums.size() == 1){
            if(abs(nums[0] - 24) < eps)
                ans = true;
            return;
        }
        for(int i = 0; i < nums.size(); i++){
            for(int j = 0; j < i; j++){
                double p = nums[i], q = nums[j];
                vector<double> tmp{p+q, p-q, q-p, p*q};
                if(abs(p) > eps)
                    tmp.push_back(q / p); //p不为0时
                if(abs(q) > eps)
                    tmp.push_back(p / q); //q不为0时
                nums.erase(nums.begin() + i); //注意释放顺序，先释放下标大的元素
                nums.erase(nums.begin() + j);
                for(double t : tmp){
                    nums.push_back(t);
                    helper(nums, eps, ans);
                    nums.pop_back();
                }
                nums.insert(nums.begin() + j); //注意插入顺序，先插入下标小的元素
                nums.insert(nums.begin() + i);
            }
        }
    }
};
```

#### Recursive Solution

* 思想与上面一致，仅仅是写法不同

```
class Solution {
public:
    bool judgePoint24(vector<int>& nums) {
        double eps = 0.001;
        vector<char> ops{'+', '-', '*', '/'};
        vector<double> arr(nums.begin(), nums.end());
        return helper(arr, ops, eps);
    }
    bool helper(vector<double>& nums, vector<char>& ops, double eps){
        if(nums.size() == 1)
            return abs(nums[0] - 24) < eps;
        for(int i = 0; i < nums.size(); i++){
            for(int j = 0; j < nums.size(); j++){
                if( i == j)
                    continue;
                vector<double> t;
                for(int k = 0; k < nums.size(); k++){
                    if(k != i && k != j)
                        t.push_back(nums[k]);
                }
                for(char op : ops){
                    if((op == '+' || op == '*') && i > j) // 避免加法和乘法的重复计算
                        continue;
                    if(op == '/' && abs(nums[j]) < eps) //分子不为0
                        continue;
                    switch(op){
                        case '+': t.push_back(nums[i] + nums[j]); break;
                        case '-': t.push_back(nums[i] - nums[j]); break;
                        case '*': t.push_back(nums[i] * nums[j]); break;
                        case '/': t.push_back(nums[i] / nums[j]); break;
                    }
                    if(helper(t, ops, eps))
                        return true;
                    t.pop_back();
                }
            }
        }
        return false;
    }
};

```

#### 输出所有的24点表达式

```

class Solution {
public:
    vector<string> judgePoint24(vector<int>& nums) {
        unordered_set<string> res;
        double eps = 0.001;
        vector<char> ops = {'+', '-', '*', '/'};
        vector<double> arr(nums.begin(), nums.end());
        vector<string> vec;
        for(int num : nums)
            vec.push_back(to_string(num));
        helper(arr, ops, eps, vec, res);
        return vector<string>(res.begin(), res.end());
    }

    void helper(vector<double>& nums, vector<char>& ops, double eps, vector<string>& vec, unordered_map<string>& res)
    if(nums.size() == 1){
        if(abs(nums[0] - 24) < eps)
            res.insert(vec[0]);
        return;
    }
    for(int i = 0; i < nums.size(); i++){
        for(int j = 0; j < nums.size(); j++){
            if(i == j)
                continue;
            vector<double> t;
            vector<string> v;
            for(int k = 0; k < nums.size(); k++){
                if(k != i && k != j){
                    t.push_back(nums[k]);
                    v.push_back(vec[k]);
                }
            }
            for(char op : ops){
                if((op == '+' || op == '*') && i > j)
                    continue;
                if(op == '/' && abs(nums[j]) < eps)
                    continue;
                switch(op){
                    case '+':
                        t.push_back(nums[i] + nums[j]);
                        v.push_back("(" + vec[i] + "+" + vec[j] + ")");
                    case '-':
                        t.push_back(nums[i] - nums[j]);
                        v.push_back("(" + vec[i] + "-" + vec[j] + ")");
                    case '*':
                        t.push_back(nums[i] * nums[j]);
                        v.push_back("(" + vec[i] + "*" + vec[j] + ")");
                    case '/':
                        t.push_back(nums[i] / nums[j]);
                        v.push_back("(" + vec[i] + "/" + vec[j] + ")");
                }
                helper(t, ops, eps, v, res);
                t.pop_back();
                v.pop_back();
            }

        }
    }
};
```
